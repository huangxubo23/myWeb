2、已知有字符串foo="get-element-by-id",写一个function将其转化成驼峰表示法"getElementById"。

var foo="get-element-by-id";
function changeUp(){
    var str = combo(foo);
    alert(str);
}
function combo(msg) {
    var arr = msg.split("-");
    for (var n = 1; n < arr.length; n++) {
        arr[n] = arr[n].charAt(0).toUpperCase() + arr[n].substr(1, arr[n].length);
    }
    msg = arr.join("");
    return msg;
}
